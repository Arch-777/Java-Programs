#include <graphics.h>
#include <dos.h>
#include <stdio.h>
#include <conio.h>

int main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "C:\\TURBO3\\BGI");

    int x1, y1, x2, y2;
    int dx, dy, p, x, y;

    printf("Enter the coordinates of the starting point (x1 y1): ");
    scanf("%d %d", &x1, &y1);
    printf("Enter the coordinates of the ending point (x2 y2): ");
    scanf("%d %d", &x2, &y2);

    dx = x2 - x1;
    dy = y2 - y1;
    p = 2 * dy - dx;
    x = x1;
    y = y1;

    for (int i = x; i <= x2; i++) {
        putpixel(x, y, WHITE);
        x++;

        if (p < 0) {
            p += 2 * dy;
        } else {
            y++;
            p += 2 * dy - 2 * dx;
        }
    }

    getch();
    closegraph();
    return 0;
}
