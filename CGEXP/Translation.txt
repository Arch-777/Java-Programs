#include <stdio.h>
#include <conio.h>
#include <graphics.h>

void main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "C:\\tc\\BGI");
    
    int tx = 50, ty = 50, x1 = 100, x2 = 230, y1 = 100, y2 = 70;
    
    cleardevice();
    printf("Rectangle before translation:\n");
    rectangle(x1, y1, x2, y2);
    getch();
    
    cleardevice();
    printf("Rectangle after translation:\n");
    rectangle(x1 + tx, y1 + ty, x2 + tx, y2 + ty);
    getch();
    
    closegraph();
}
