#include<stdio.h>
#include<conio.h>
#include<graphics.h>
#include<math.h>
#include<dos.h> void
main()
{
int gd=DETECT,gm; int
x1,x2,y1,y2; int
dx,dy,length,i; float
xinc,yinc,xnew,ynew;
clrscr();
initgraph(&gd,&gm,"C:\TURBOC3\BGI");
printf("Enter the co-ordinate of x1:");
scanf("%d",&x1);
printf("Enter the co-ordinate of y1:");
scanf("%d",&y1);
printf("Enter the co-ordinate of x2:");
scanf("%d",&x2);
printf("Enter the co-ordinate of y2:");
scanf("%d",&y2); dx=abs(x2-
x1);
dy=abs(y2-y1);
if(dx>=dy)
{
length=dx;
}
else
{
length=dy;
}
xinc=dx/length;
yinc=dy/length;
xnew=x1;
ynew=y1;
for(i=0;i<length;i++)
{
xnew=xnew+xinc;
ynew=ynew+yinc;
putpixel(xnew,ynew,25);
}

}
