#include <stdio.h>
#include <conio.h>
#include <graphics.h>

void main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "C:\\tc\\BGI");

    int x1 = 50, y1 = 150, x2 = 75, y2 = 125, x3 = 100, y3 = 150;

    cleardevice();
    printf("Triangle before reflection:\n");
    line(x1, y1, x2, y2);
    line(x1, y1, x3, y3);
    line(x2, y2, x3, y3);
    getch();

    cleardevice();
    printf("Triangle after reflection over x-axis:\n");
    line(x1, -y1 + 200, x2, -y2 + 200);
    line(x1, -y1 + 200, x3, -y3 + 200);
    line(x2, -y2 + 200, x3, -y3 + 200);
    getch();

    closegraph();
}
