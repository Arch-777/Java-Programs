#include <stdio.h>
#include <conio.h>
#include <graphics.h>

void main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "C:\\tc\\BGI");

    int x1 = 30, y1 = 30, x2 = 70, y2 = 70;
    int sx = 2, sy = 2;

    cleardevice();
    printf("Rectangle before scaling:\n");
    rectangle(x1, y1, x2, y2);
    getch();

    cleardevice();
    printf("Rectangle after scaling:\n");
    rectangle(x1 * sx, y1 * sy, x2 * sx, y2 * sy);
    getch();

    closegraph();
}
