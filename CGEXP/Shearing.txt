#include <stdio.h>
#include <conio.h>
#include <graphics.h>

void main() {
    int gd = DETECT, gm;
    initgraph(&gd, &gm, "C:\\tc\\BGI");

    int x1 = 100, x2 = 100, y1 = 100, y2 = 30, x3 = 170, y3 = 30, x4 = 170, y4 = 100;
    int shx = 1; 

    cleardevice();
    printf("Rectangle before shearing:\n");
    line(x1, y1, x2, y2);
    line(x1, y1, x4, y4);
    line(x2, y2, x3, y3);
    line(x3, y3, x4, y4);
    getch();

    cleardevice();
    printf("Rectangle after shearing:\n");
    line(x1 + shx * y1, y1, x2 + shx * y2, y2);
    line(x1 + shx * y1, y1, x4 + shx * y4, y4);
    line(x2 + shx * y2, y2, x3 + shx * y3, y3);
    line(x3 + shx * y3, y3, x4 + shx * y4, y4);
    getch();

    closegraph();
}
