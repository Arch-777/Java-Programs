#include<stdio.h>
#include<graphics.h>
#include<conio.h>
#include<stdlib.h>
int main()
{
 int gd = DETECT, gm, n, *x, i, k = 0;
 int wx1 = 220, wy1 = 140, wx2 = 420, wy2 = 140, wx3 = 420, wy3 = 340, wx4 = 220, wy4 = 340;

 printf("Window:-");
 setcolor(RED);
 drawpoly(5, w);

 scanf("%d", &n);
 x = (int*) malloc((n * 2 + 2) * sizeof(int));
 printf("Enter the coordinates of points:\n");
 k = 0;
 for (i = 0; i < n * 2; i += 2)
{
 printf("(x%d, y%d): ", k, k);
 scanf("%d %d", &x[i], &x[i + 1]);
 k++;
 }

 cleardevice();
 closegraph();
 free(x);
 return 0;
}
