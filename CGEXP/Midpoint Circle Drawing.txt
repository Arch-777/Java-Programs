#include<stdio.h> 
#include<conio.h>
#include<stdlib.h> 
#include <math.h>
#include <graphics.h>
int main() 
{ int gd = DETECT, gm;
int xc, yc, r;
int x = 0, y = r;
int Pk = 1 - r; 
initgraph(&gd, &gm, ""); 
printf("Enter Value Of X: "); 
scanf("%d", &xc); 
printf("Enter Value Of Y: ");
scanf("%d", &yc); 
printf("Enter The Radius Of The Circle: "); 
scanf("%d", &r); 
while (x < y) {
x++; 
if (Pk < 0) {
Pk += 2 * x + 1;
} else 
{ y--; Pk += 2 * (x - y) + 1;
} 
putpixel(xc + x, yc + y, 5);
putpixel(xc - x, yc + y, 5);
putpixel(xc + x, yc - y, 5);
putpixel(xc - x, yc - y, 5); 
putpixel(xc + y, yc + x, 5); 
putpixel(xc - y, yc + x, 5); 
putpixel(xc + y, yc - x, 5);
putpixel(xc - y, yc - x, 5); }
getch();
return 0; }
