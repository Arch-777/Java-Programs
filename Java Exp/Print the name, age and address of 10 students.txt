public class Student {
    private String name;
    private int age;
    private String address;

    // Default constructor
    public Student() {
        name = "unknown";
        age = 0;
        address = "not available";
    }

    // Method overloading: setInfo with two parameters
    public void setInfo(String nameParam, int ageParam) {
        name = nameParam;
        age = ageParam;
    }

    // Method overloading: setInfo with three parameters
    public void setInfo(String nameParam, int ageParam, String addressParam) {
        name = nameParam;
        age = ageParam;
        address = addressParam;
    }

    // Method to print student information
    public void printInfo() {
        System.out.println("Name: " + name);
        System.out.println("Age: " + age);
        System.out.println("Address: " + address);
        System.out.println();
    }

    public static void main(String[] args) {
        // Create an array of 10 Student objects
        Student[] students = new Student[10];

        // Initialize and set info for each student
        for (int i = 0; i < students.length; i++) {
            students[i] = new Student();
            if (i % 2 == 0) {
                students[i].setInfo("Student" + (i + 1), 20 + i);
            } else {
                students[i].setInfo("Student" + (i + 1), 20 + i, "Address" + (i + 1));
            }
        }

        // Print information for all students
        for (Student student : students) {
            student.printInfo();
        }
    }
}
