class OverloadingDemo
{
void add(int a,int b)
{System.out.println("Add with 2 parameter called: "+ (a+b));}

void add(int a,int b,int c)
{System.out.println("Add with 3 parameter called: "+ (a+b+c));}

void add(int a,int b,int c,int d)
{System.out.println("Add with 4 parameter called: " + (a+b+c+d));}

public static void main(String[] args)
{ OverloadingDemo o=new OverloadingDemo();
o.add(5,4);
o.add(5,4,6);
o.add(3,4,6,7);
}}
